#include <iostream>
using namespace std;

void find_triplet_indices(int arr[], int n) {
    for (int k = n - 1; k >= 0; --k) {
        int i = 0;
        int j = k - 1;
        while (i < j) {
            int sum = arr[i] + arr[j];
            if (sum == arr[k]) {
                cout << i+1 << ", " << j+1 << ", " << k+1 << endl;
                return;
            } else if (sum < arr[k]) {
                ++i;
            } else {
                --j;
            }
        }
    }

    cout << "No sequence found" << endl;
}

int main() {
    int T;
    cin >> T;

    for (int t = 0; t < T; ++t) {
        int n;
        cin >> n;
        int arr[n];
        for (int i = 0; i < n; ++i) {
            cin >> arr[i];
        }

        find_triplet_indices(arr, n);
    }

    return 0;
}
