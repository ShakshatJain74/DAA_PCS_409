#include <iostream>
using namespace std;

int count_pairs(int arr[], int n, int key) {
    int count = 0;
    for(int i = 0; i < n; i++) {
        for(int j = i + 1; j < n; j++) {
            if(key == abs(arr[i] - arr[j])) {
                count++;
            }
        }
    }
    return count;
}

int main() {
    int T;
    cin >> T;
    while (T--) {
        int n;
        cin >> n;
        int arr[n];
        for (int i = 0; i < n; i++) {
            cin >> arr[i];
        }
        int key;
        cin >> key;
        int c = count_pairs(arr, n, key);
        if(c) {
            cout << c << endl;
        }
        else {
            cout << "Key not present" << endl;
        }
    }
    return 0;
}
